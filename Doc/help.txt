run :
	// instruction to run
	./mincamlc.sh <yourfile.ml>


options : 
	-o <File> : to give have the output in File.s
		( give an other name of the input file. )
	-h : to print help
	-v : the version ( 1 = 1er rendu
			   2 = 2eme rendu )
	-t : to stop after the type checking
	-asml : to print the asml file.
	-verbose : to print the trace of the file
